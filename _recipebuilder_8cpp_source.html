47void URecipebuilder::CreateRecipe()

49 int numIngredients = FMath::RandRange(1, 10);
50 while(numIngredients > 0)
51 {
53 FName Ingredient = Ingredients[FMath::RandRange(0, 5)];

71void URecipebuilder::UpdateRecipeUI()
72{
74 UDetectionComponent* DetectionComponent = GetOwner()->FindComponentByClass<UDetectionComponent>();
75
76 //<Get the map of vegetables that the player actually brought in
77 TMap<FName, int>& DetectionMap = DetectionComponent->GetMap();
78
79 for(auto& p : RecipeMap)
80 {
81 //<if the veggie exists in the player's map
82 if(DetectionMap.Contains(p.Key))
83 {
84 int count = 0, i = 0;
85 //<if the player has more of a veggie
86 if(*DetectionMap.Find(p.Key) > p.Value)
87 {
88 count = p.Value;
89 DetectionMap[p.Key] = DetectionMap[p.Key] - p.Value;
90 p.Value = 0;
91 }
92 //<if the recipe has more of a veggie
93 else
94 {
95 count = DetectionMap[p.Key];
96 p.Value = p.Value - DetectionMap[p.Key];
97 DetectionMap[p.Key] = 0;
98 }
99
100 //<Despawn veggies that have been consumed
101 TSet<TWeakObjectPtr<AActor>>& VeggieSet = DetectionComponent->GetVegetableSet();
102 if(count > 0)
103 {
104 for(auto& elem : VeggieSet)
105 {
106 if(elem.IsValid() && elem->FindComponentByClass<UStaticMeshComponent>()->ComponentTags[0] == p.Key)
107 {
108 elem->Destroy();
109 if(++i >= count)break;
110 }
111 }
112 DetectionComponent->Cleanup();
113 }
114
117
118 }
119 UE_LOG(LogTemp, Warning, TEXT("KEY: %s"), *p.Key.ToString());
120 UE_LOG(LogTemp, Warning, TEXT("VALUE: %d"), p.Value);
121 }
122}
