35void UDetectionComponent::AddVegetable(AActor* Actor)
36{
37 if(!Actor->ActorHasTag("Vegetable")) return;
38 bool bAlreadyInSet = false;
39 //checks whether an element is already in the set and sets the bool to true or else adds the element to the set
40 VegetableSet.Add(Actor, &bAlreadyInSet);
41 if(bAlreadyInSet == false)
42 {
43 UE_LOG(LogTemp, Warning, TEXT("Adding..."));
44 UStaticMeshComponent* StaticMesh = Actor->FindComponentByClass<UStaticMeshComponent>();
45 if(StaticMesh!=nullptr && !StaticMesh->ComponentTags.IsEmpty())
46 {
47 const FName& Tag = StaticMesh->ComponentTags[0];
48 UE_LOG(LogTemp, Display, TEXT("%s"), *StaticMesh->ComponentTags[0].ToString());
49 if(map.Contains(Tag))++map[Tag];
50 else map.Add(Tag, 1);
51 }
52 }
53 Cleanup();
54}

56void UDetectionComponent::Cleanup()
57{
58 if(VegetableSet.Num() > 4)
59 {
60 TArray<TWeakObjectPtr<AActor>> ToRemove;
61
62 //Iterating over the set and adding invalid object references to the remove array to ensure iterator stability
63 //Don't want to remove objects as we iterate over a container to not mess up the iterators
64 for(auto& item : VegetableSet)
65 {
66 if(!item.IsValid())ToRemove.Add(item);
67 }
68
69 for(auto& item : ToRemove)
70 {
71 VegetableSet.Remove(item);
72 }
73 }
74}
